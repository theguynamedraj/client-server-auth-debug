================ LOGIN ================
curl -X POST http://localhost:3000/auth/login -H "Content-Type: application/json" -d "{\"email\":\"test@gmail.com\",\"password\":\"password123\"}"

{"message":"OTP sent","loginSessionId":"erng6k"}

================ VERIFY OTP ================
curl -c cookies.txt -X POST http://localhost:3000/auth/verify-otp -H "Content-Type: application/json" -d "{\"loginSessionId\":\"erng6k\",\"otp\":\"<YOUR_OTP>\"}"

{"message":"OTP verified","sessionId":"erng6k"}

================ GET TOKEN ================
curl -b cookies.txt -X POST http://localhost:3000/auth/token

{"access_token":"...","expires_in":900}
